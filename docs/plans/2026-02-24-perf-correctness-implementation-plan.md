# Performance & Correctness Test Suite Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a blackbox property-based correctness test suite with a test plugin running as an external process, controllable via Ergo messages for failure injection and cloud state manipulation.

**Architecture:** A standalone test plugin binary (using the plugin SDK) with a TestController actor for test harness control. The test harness is a Go test binary with its own Ergo node for sending control messages to the TestController, and a REST client for interacting with the formae agent. Operations are generated by `rapid` and dispatched either via REST (user ops) or Ergo (cloud/fault ops).

**Tech Stack:** Go, Ergo actor framework, `pgregory.net/rapid` for property-based testing, formae plugin SDK, formae REST API client.

**Design Doc:** `docs/plans/2026-02-21-performance-and-correctness-testing-design.md`

---

## Decisions Made

These resolve open questions from the design doc:

1. **Test plugin location:** `tests/testplugin/` — it's test infrastructure, not a shipped plugin.
2. **Agent lifecycle:** In-process metastructure + HTTP server (like distributed workflow tests). The test plugin runs as an external binary. This gives us REST API access while keeping agent setup simple.
3. **Test plugin namespace:** `Test` with resource type `Test::Generic::Resource`. Simple, focused on orchestration testing.
4. **Test plugin entry point:** Custom `main()` that replicates `plugin.Run()` logic plus spawns TestController. This avoids modifying the plugin SDK.
5. **Blackbox test location:** `tests/blackbox/` in main Go module with `//go:build blackbox` tag.

```go
package main

import "sync"

type CloudStateEntry struct {
    NativeID     string
    ResourceType string
    Properties   string // JSON
}

type CloudState struct {
    mu      sync.RWMutex
    entries map[string]CloudStateEntry // keyed by NativeID
}

func NewCloudState() *CloudState {
    return &CloudState{
        entries: make(map[string]CloudStateEntry),
    }
}

func (cs *CloudState) Put(nativeID, resourceType, properties string) {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    cs.entries[nativeID] = CloudStateEntry{
        NativeID:     nativeID,
        ResourceType: resourceType,
        Properties:   properties,
    }
}

func (cs *CloudState) Get(nativeID string) (CloudStateEntry, bool) {
    cs.mu.RLock()
    defer cs.mu.RUnlock()
    entry, ok := cs.entries[nativeID]
    return entry, ok
}

func (cs *CloudState) Delete(nativeID string) {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    delete(cs.entries, nativeID)
}

func (cs *CloudState) ListNativeIDs(resourceType string) []string {
    cs.mu.RLock()
    defer cs.mu.RUnlock()
    var ids []string
    for _, entry := range cs.entries {
        if entry.ResourceType == resourceType {
            ids = append(ids, entry.NativeID)
        }
    }
    return ids
}

func (cs *CloudState) Snapshot() map[string]CloudStateEntry {
    cs.mu.RLock()
    defer cs.mu.RUnlock()
    snap := make(map[string]CloudStateEntry, len(cs.entries))
    for k, v := range cs.entries {
        snap[k] = v
    }
    return snap
}
```

```go
package main

import (
    "sync"
    "time"
)

type OperationLog struct {
    mu      sync.Mutex
    entries []OperationLogEntry
}

func NewOperationLog() *OperationLog {
    return &OperationLog{}
}

func (l *OperationLog) Record(op, resourceType, nativeID string, success bool, errMsg string, duration time.Duration) {
    l.mu.Lock()
    defer l.mu.Unlock()
    l.entries = append(l.entries, OperationLogEntry{
        Timestamp:    time.Now(),
        Operation:    op,
        ResourceType: resourceType,
        NativeID:     nativeID,
        Success:      success,
        Error:        errMsg,
        Duration:     duration,
    })
}

func (l *OperationLog) Snapshot() []OperationLogEntry {
    l.mu.Lock()
    defer l.mu.Unlock()
    snap := make([]OperationLogEntry, len(l.entries))
    copy(snap, l.entries)
    return snap
}
```

**Step 3: Implement TestController actor**

File: `tests/testplugin/test_controller.go`

```go
package main

import (
    "fmt"

    "ergo.services/ergo/act"
    "ergo.services/ergo/gen"
)

const TestControllerName = gen.Atom("TestController")

type InjectionRule struct {
    Operation    string
    ResourceType string
    ErrorMessage string
    Remaining    int // 0 = permanent
}

type LatencyRule struct {
    Operation    string
    ResourceType string
    Duration     time.Duration
}

type TestController struct {
    act.Actor

    cloudState   *CloudState
    opLog        *OperationLog
    errorRules   []InjectionRule
    latencyRules []LatencyRule
}

func NewTestController() gen.ProcessBehavior {
    return &TestController{}
}

func (tc *TestController) Init(args ...any) error {
    cs, ok := tc.Env("CloudState")
    if !ok {
        return fmt.Errorf("TestController: missing 'CloudState' env")
    }
    tc.cloudState = cs.(*CloudState)

    ol, ok := tc.Env("OperationLog")
    if !ok {
        return fmt.Errorf("TestController: missing 'OperationLog' env")
    }
    tc.opLog = ol.(*OperationLog)

    tc.Log().Info("TestController started")
    return nil
}

func (tc *TestController) HandleCall(from gen.PID, ref gen.Ref, message any) (any, error) {
    switch msg := message.(type) {
    case GetCloudStateSnapshot:
        return CloudStateSnapshotResponse{Entries: tc.cloudState.Snapshot()}, nil
    case GetOperationLog:
        return OperationLogResponse{Entries: tc.opLog.Snapshot()}, nil
    case InjectError:
        tc.errorRules = append(tc.errorRules, InjectionRule{
            Operation:    msg.Operation,
            ResourceType: msg.ResourceType,
            ErrorMessage: msg.ErrorMessage,
            Remaining:    msg.Count,
        })
        return Ack{}, nil
    case InjectLatency:
        tc.latencyRules = append(tc.latencyRules, LatencyRule{
            Operation:    msg.Operation,
            ResourceType: msg.ResourceType,
            Duration:     msg.Duration,
        })
        return Ack{}, nil
    case ClearInjections:
        tc.errorRules = nil
        tc.latencyRules = nil
        return Ack{}, nil

    // Cloud state manipulation
    case ModifyCloudState:
        entry, ok := tc.cloudState.Get(msg.NativeID)
        if ok {
            tc.cloudState.Put(msg.NativeID, entry.ResourceType, msg.Properties)
        }
        return Ack{}, nil
    case DeleteCloudState:
        tc.cloudState.Delete(msg.NativeID)
        return Ack{}, nil
    case CreateCloudState:
        tc.cloudState.Put(msg.NativeID, msg.ResourceType, msg.Properties)
        return Ack{}, nil

    default:
        return nil, fmt.Errorf("TestController: unknown message type: %T", msg)
    }
}

// CheckError checks if an injected error rule matches and returns the error.
// Called by the plugin's CRUD methods before executing.
func (tc *TestController) CheckError(operation, resourceType string) error {
    for i, rule := range tc.errorRules {
        if rule.Operation == operation && (rule.ResourceType == "" || rule.ResourceType == resourceType) {
            if rule.Remaining > 0 {
                tc.errorRules[i].Remaining--
                if tc.errorRules[i].Remaining == 0 {
                    // Remove exhausted rule
                    tc.errorRules = append(tc.errorRules[:i], tc.errorRules[i+1:]...)
                }
            }
            return fmt.Errorf("%s", rule.ErrorMessage)
        }
    }
    return nil
}

// CheckLatency returns the latency to inject for the given operation.
func (tc *TestController) CheckLatency(operation, resourceType string) time.Duration {
    for _, rule := range tc.latencyRules {
        if rule.Operation == operation && (rule.ResourceType == "" || rule.ResourceType == resourceType) {
            return rule.Duration
        }
    }
    return 0
}
```

Note: The TestController's `CheckError`/`CheckLatency` methods are called by the plugin CRUD methods. Since both run in the same process, direct method calls are safe here (no actor boundary crossing — the CloudState and TestController are shared in-process state).

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

type InjectionRule struct {
    Operation    string
    ResourceType string
    ErrorMessage string
    Remaining    int
}

type LatencyRule struct {
    Operation    string
    ResourceType string
    Duration     time.Duration
}

type InjectionState struct {
    mu           sync.Mutex
    errorRules   []InjectionRule
    latencyRules []LatencyRule
}

func NewInjectionState() *InjectionState {
    return &InjectionState{}
}

func (is *InjectionState) AddError(operation, resourceType, errorMessage string, count int) {
    is.mu.Lock()
    defer is.mu.Unlock()
    is.errorRules = append(is.errorRules, InjectionRule{
        Operation:    operation,
        ResourceType: resourceType,
        ErrorMessage: errorMessage,
        Remaining:    count,
    })
}

func (is *InjectionState) AddLatency(operation, resourceType string, duration time.Duration) {
    is.mu.Lock()
    defer is.mu.Unlock()
    is.latencyRules = append(is.latencyRules, LatencyRule{
        Operation:    operation,
        ResourceType: resourceType,
        Duration:     duration,
    })
}

func (is *InjectionState) Clear() {
    is.mu.Lock()
    defer is.mu.Unlock()
    is.errorRules = nil
    is.latencyRules = nil
}

func (is *InjectionState) CheckError(operation, resourceType string) error {
    is.mu.Lock()
    defer is.mu.Unlock()
    for i, rule := range is.errorRules {
        if rule.Operation == operation && (rule.ResourceType == "" || rule.ResourceType == resourceType) {
            if rule.Remaining > 0 {
                is.errorRules[i].Remaining--
                if is.errorRules[i].Remaining == 0 {
                    is.errorRules = append(is.errorRules[:i], is.errorRules[i+1:]...)
                }
            }
            return fmt.Errorf("%s", rule.ErrorMessage)
        }
    }
    return nil
}

func (is *InjectionState) CheckLatency(operation, resourceType string) time.Duration {
    is.mu.Lock()
    defer is.mu.Unlock()
    for _, rule := range is.latencyRules {
        if rule.Operation == operation && (rule.ResourceType == "" || rule.ResourceType == resourceType) {
            return rule.Duration
        }
    }
    return 0
}
```

```go
func OperationSequenceGen(config PropertyTestConfig) *rapid.Generator[[]Operation] {
    return rapid.Custom(func(t *rapid.T) []Operation {
        count := rapid.IntRange(config.OperationCount.Min, config.OperationCount.Max).Draw(t, "count")
        ops := make([]Operation, count)
        for i := range ops {
            ops[i] = SingleOperationGen(config).Draw(t, fmt.Sprintf("op-%d", i))
        }
        return ops
    })
}

func SingleOperationGen(config PropertyTestConfig) *rapid.Generator[Operation] {
    return rapid.Custom(func(t *rapid.T) Operation {
        // Build list of allowed operation kinds based on config
        kinds := []OperationKind{OpApply, OpDestroy, OpVerifyState, OpTriggerSync}
        if config.EnableFailures {
            kinds = append(kinds, OpInjectError, OpClearInjections)
        }
        if config.EnableCloudChanges {
            kinds = append(kinds, OpCloudModify, OpCloudDelete)
        }
        // ... etc

        kindIdx := rapid.IntRange(0, len(kinds)-1).Draw(t, "kind")
        kind := kinds[kindIdx]

        // Generate resource IDs
        resourceCount := rapid.IntRange(1, config.ResourceCount).Draw(t, "resCount")
        resourceIDs := rapid.SliceOfN(
            rapid.IntRange(0, config.ResourceCount-1),
            resourceCount, resourceCount,
        ).Draw(t, "resourceIDs")

        return Operation{
            Kind:        kind,
            ResourceIDs: resourceIDs,
            ApplyMode:   applyModeGen(t),
            Blocking:    !config.EnableConcurrency || rapid.Bool().Draw(t, "blocking"),
        }
    })
}
```
